    def expression_for(arg)
      if arg.is_a? String
        "#{self.field}#{self.op}('#{arg}')"
      elsif arg.is_a? Regexp
        "#{self.field}#{self.op}(#{arg.inspect})"
      elsif arg.is_a? Date
        "#{self.field}#{self.op}(DateTime.parse(\"#{arg}\"))"
      else
        "#{self.field}#{self.op}(#{arg})"        
      end
    end
