    def update_attributes(hash)
      hash.each_key do |k|
        if hash[k].is_a? Hash
          hash[k].each do |k,v|
            self.send(k.to_s+"=",v)
          end
        else
          self.send(k.to_s+"=", hash[k])
        end
      end
      self.save
    end
